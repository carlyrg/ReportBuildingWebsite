% Admitted

Description: Percentage of individuals with a [Lead Score Master] = "Move In" or "Move Out" compared to number of referrals.

% Admitted =
DIVIDE ( [Admit], [Referrals], 0 )

% Denied

Description: Percentage of individual referrals that have a [Lead Score Master] = "Unqualified" compared to number of referrals.

% Denied =
DIVIDE ( [Denied], [Referrals], 0 )

% Lost

Description: Percentage of individual referrals that resulted in [Lead Score Master] of "Lost Lead"

% Lost =
DIVIDE ( [Lost], [Referrals], 0 )

% Pending

Description: Percentage of individuals with no outcome compared to number of referrals.

% Pending =
DIVIDE ( [Pending], [Referrals], 0 )

Accepted

Description: Number of individuals with a [Lead Score Master] = "Accepted".

Accepted =
VAR Outcomes =
    CALCULATETABLE (
        VALUES ( 'Individual Dimension'[Individual Lead Score Master] ),
        KEEPFILTERS ( 'Individual Dimension'[Individual Lead Score Master]
            IN { "Move In", "Lost Lead", "Unqualified", "Move Out" } )
    )
RETURN
    SUMX (
        CALCULATETABLE (
            SUMMARIZE (
                'Individual Facts',
                'Individual Dimension'[GroupIndividualLinkKey],
                'Individual Dimension'[UserLocalIndividualInquiryDate]
            ),
            USERELATIONSHIP ( 'Individual Dimension'[UserLocalIndividualInquiryDate], 'Date Dimension'[Date] )
        ),
        IF (
            ISBLANK (
                COUNTROWS (
                    CALCULATETABLE (
                        VALUES ( 'Individual Dimension'[Individual Lead Score Master] ),
                        'Individual Dimension'[Individual Lead Score Master]
                            IN { "Move In", "Move Out" }
                    )
                )
            ),
            IF (
                COUNTROWS (
                    CALCULATETABLE (
                        VALUES ( 'Individual Dimension'[Individual Lead Score Master] ),
                        'Individual Dimension'[Individual Lead Score Master] = "Accepted"
                    )
                ) >= 1,
                1
            )
        )
    )

Active Prospects

Description: Number of prospects that have an active status.

Active Prospects =
VAR MaxDate =
    MAX ( 'Date Dimension'[Date] )
VAR ActiveProspects =
    CALCULATETABLE (
        VALUES ( 'Individual Dimension'[IndividualId] ),
        'Individual History Dimension'[Individual Start Effective DateTime] <= MaxDate,
        'Individual History Dimension'[Individual End Effective DateTime] >= MaxDate
            || ISBLANK ( 'Individual History Dimension'[Individual End Effective DateTime] ),
        'Individual History Dimension'[Individual Historic Status] = "Active",
        'Individual History Dimension'[Individual Historic Type] = "Prospect",
        CROSSFILTER ( 'Individual History Dimension'[IndividualId], 'Individual Dimension'[IndividualId], BOTH ),
        'Individual Facts'
    )
RETURN
    COUNTROWS ( ActiveProspects )

Admit

Admit =
SUMX (
    SUMMARIZE (
        'Individual Facts',
        'Individual Dimension'[GroupIndividualLinkKey],
        'Individual Dimension'[UserLocalIndividualInquiryDate]
    ),
    IF (
        CALCULATE (
            COUNTROWS (
                CALCULATETABLE (
                    VALUES ( 'Individual Dimension'[Individual Lead Score Master] ),
                    CROSSFILTER ( 'Individual Facts'[IndividualId], 'Individual Dimension'[IndividualId], BOTH ),
                    KEEPFILTERS ( 'Individual Dimension'[Individual Lead Score Master]
                        IN { "Move In", "Move Out" } )
                )
            ),
            USERELATIONSHIP ( 'Individual Dimension'[Outcome Date], 'Date Dimension'[Date] )
        ) >= 1,
        1

Availability

Description: Number of units that have no active [Resident] and no [Upcoming Individual Move Ins]

Availability =
VAR MaxDate =
    MAX ( 'Date Dimension'[Date] )
VAR AllUnits =
    CALCULATETABLE (
        VALUES ( 'Unit Historical Facts'[UnitKey] ),
        'Unit Dimension'[Unit Start Effective DateTime] <= MaxDate,
        'Unit Dimension'[Unit End Effective DateTime] >= MaxDate
            || ISBLANK ( 'Unit Dimension'[Unit End Effective DateTime] ),
        'Unit Dimension'[CountTowardsTotal] = 1
    )
VAR OccupiedUnits =
    CALCULATETABLE (
        VALUES ( 'Housing Contract Facts'[UnitKey] ),
        CROSSFILTER ( 'Housing Contract Facts'[MoveInDate], 'Date Dimension'[Date], NONE ),
        'Housing Contract Facts'[MoveInDate] <= MaxDate,
        'Housing Contract Facts'[MoveOutDate] >= MaxDate
            || ISBLANK ( 'Housing Contract Facts'[MoveOutDate] ),
        ISBLANK ( 'Housing Contract Facts'[CancelledDate] ),
        'Unit Dimension'[CountTowardsTotal] = 1
    )
VAR VacantUnits =
    EXCEPT ( AllUnits, OccupiedUnits )
VAR PendingMoveOuts =
    CALCULATETABLE (
        VALUES ( 'Housing Contract Facts'[UnitKey] ),
        CROSSFILTER ( 'Housing Contract Facts'[MoveInDate], 'Date Dimension'[Date], NONE ),
        NOT ( ISBLANK ( 'Housing Contract Facts'[MoveOutDate] ) ),
        'Housing Contract Facts'[MoveOutDate] > MaxDate,
        ISBLANK ( 'Housing Contract Facts'[CancelledDate] )
    )
VAR AvailableLessPendingMoveIns =
    DISTINCT ( UNION ( VacantUnits, PendingMoveOuts ) )
VAR PendingMoveIns =
    CALCULATETABLE (
        VALUES ( 'Housing Contract Facts'[UnitKey] ),
        CROSSFILTER ( 'Housing Contract Facts'[MoveInDate], 'Date Dimension'[Date], NONE ),
        'Housing Contract Facts'[MoveInDate] > MaxDate,
        NOT ( ISBLANK ( 'Housing Contract Facts'[MoveInDate] ) ),
        ISBLANK ( 'Housing Contract Facts'[CancelledDate] )
    )
VAR Availability =
    EXCEPT ( AvailableLessPendingMoveIns, PendingMoveIns )
RETURN
    COUNTROWS ( Availability )
